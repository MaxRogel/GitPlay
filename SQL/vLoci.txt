SELECT vAreas.Area_ID, vAreas.YYYY, vAreas.AreaName, Loci.Locus_ID, Loci.Locus_no, Loci.Square, Loci.Date_opened, Loci.Date_closed, Loci.Open_Level, Loci.Close_Level, Loci.Locus_Above, Loci.Locus_Below, Loci.Locus_CoExisting, Loci.Description, Loci.Deposit_description, Loci.Registration_notes, Loci.Clean
FROM vAreas INNER JOIN Loci ON vAreas.Area_ID = Loci.Area_ID
ORDER BY vAreas.YYYY, vAreas.AreaName, Loci.Locus_no;
